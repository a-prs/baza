# Мультиязычный бот (i18n)

Твой бот заговорил на нескольких языках — это открывает международную аудиторию. Разберём как это сделать без лишних библиотек, а потом покажем с Babel.

## Простой подход: словарь переводов

Для 2–3 языков достаточно простого словаря:

```python
# translations.py
TRANSLATIONS = {
    "ru": {
        "welcome": "Привет, {name}! Добро пожаловать.",
        "help": "Доступные команды:\n/start — начало\n/help — помощь\n/lang — язык",
        "lang_changed": "Язык изменён на русский.",
        "choose_lang": "Выберите язык:",
        "unknown": "Не понимаю. Попробуй /help",
        "buy": "Купить",
        "cancel": "Отмена",
    },
    "en": {
        "welcome": "Hello, {name}! Welcome.",
        "help": "Available commands:\n/start — start\n/help — help\n/lang — language",
        "lang_changed": "Language changed to English.",
        "choose_lang": "Choose language:",
        "unknown": "I don't understand. Try /help",
        "buy": "Buy",
        "cancel": "Cancel",
    },
    "de": {
        "welcome": "Hallo, {name}! Willkommen.",
        "help": "Verfügbare Befehle:\n/start — Start\n/help — Hilfe\n/lang — Sprache",
        "lang_changed": "Sprache auf Deutsch geändert.",
        "choose_lang": "Sprache wählen:",
        "unknown": "Ich verstehe nicht. Versuche /help",
        "buy": "Kaufen",
        "cancel": "Abbrechen",
    }
}

DEFAULT_LANG = "ru"
SUPPORTED_LANGS = list(TRANSLATIONS.keys())


def t(key: str, lang: str = DEFAULT_LANG, **kwargs) -> str:
    """Перевести ключ на язык. kwargs — для форматирования."""
    text = TRANSLATIONS.get(lang, TRANSLATIONS[DEFAULT_LANG]).get(key, key)
    if kwargs:
        text = text.format(**kwargs)
    return text
```

## Хранение языка пользователя

```python
import sqlite3
import os

DB_PATH = os.getenv("DB_PATH", "bot.db")

def init_db():
    conn = sqlite3.connect(DB_PATH)
    conn.execute("""
        CREATE TABLE IF NOT EXISTS user_settings (
            user_id INTEGER PRIMARY KEY,
            lang TEXT DEFAULT 'ru',
            updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    """)
    conn.commit()
    conn.close()

def get_lang(user_id: int) -> str:
    conn = sqlite3.connect(DB_PATH)
    row = conn.execute(
        "SELECT lang FROM user_settings WHERE user_id = ?", (user_id,)
    ).fetchone()
    conn.close()
    return row[0] if row else DEFAULT_LANG

def set_lang(user_id: int, lang: str):
    conn = sqlite3.connect(DB_PATH)
    conn.execute("""
        INSERT INTO user_settings (user_id, lang) VALUES (?, ?)
        ON CONFLICT(user_id) DO UPDATE SET lang = ?
    """, (user_id, lang, lang))
    conn.commit()
    conn.close()
```

## Middleware для автоматической подстановки языка

```python
from aiogram import BaseMiddleware
from aiogram.types import TelegramObject, User

class I18nMiddleware(BaseMiddleware):
    async def __call__(self, handler, event: TelegramObject, data: dict):
        # Достать пользователя из события
        user: User | None = data.get("event_from_user")
        
        if user:
            lang = get_lang(user.id)
            data["lang"] = lang
            # Удобная функция перевода с фиксированным языком
            data["_"] = lambda key, **kw: t(key, lang, **kw)
        
        return await handler(event, data)
```

Зарегистрировать middleware:
```python
dp.message.middleware(I18nMiddleware())
dp.callback_query.middleware(I18nMiddleware())
```

## Хэндлеры с i18n

```python
from aiogram import Bot, Dispatcher, F
from aiogram.types import Message, CallbackQuery, InlineKeyboardMarkup, InlineKeyboardButton
from aiogram.filters import Command
from typing import Callable

bot = Bot(token=os.getenv("BOT_TOKEN"))
dp = Dispatcher()


def lang_keyboard() -> InlineKeyboardMarkup:
    lang_names = {"ru": "🇷🇺 Русский", "en": "🇬🇧 English", "de": "🇩🇪 Deutsch"}
    return InlineKeyboardMarkup(inline_keyboard=[[
        InlineKeyboardButton(text=name, callback_data=f"setlang:{code}")
        for code, name in lang_names.items()
    ]])


@dp.message(Command("start"))
async def cmd_start(message: Message, _: Callable):
    await message.answer(_("welcome", name=message.from_user.first_name))


@dp.message(Command("help"))
async def cmd_help(message: Message, _: Callable):
    await message.answer(_("help"))


@dp.message(Command("lang"))
async def cmd_lang(message: Message, _: Callable):
    await message.answer(_("choose_lang"), reply_markup=lang_keyboard())


@dp.callback_query(F.data.startswith("setlang:"))
async def handle_set_lang(callback: CallbackQuery):
    lang = callback.data.split(":")[1]
    
    if lang not in SUPPORTED_LANGS:
        await callback.answer("Unknown language")
        return
    
    set_lang(callback.from_user.id, lang)
    # Получить перевод на НОВЫЙ язык
    await callback.message.edit_text(t("lang_changed", lang))
    await callback.answer()


@dp.message(F.text & ~F.text.startswith("/"))
async def handle_unknown(message: Message, _: Callable):
    await message.answer(_("unknown"))
```

## Автоопределение языка

Telegram передаёт `language_code` пользователя. Используй при первом старте:

```python
@dp.message(Command("start"))
async def cmd_start(message: Message, _: Callable, lang: str):
    # Если язык не задан (первый старт) — определить из Telegram
    if lang == DEFAULT_LANG:  # дефолт = ещё не выбрал
        tg_lang = message.from_user.language_code or "ru"
        
        # Маппинг: Telegram code → наш код
        lang_map = {"ru": "ru", "en": "en", "de": "de", "uk": "ru"}
        detected = lang_map.get(tg_lang[:2], DEFAULT_LANG)
        
        set_lang(message.from_user.id, detected)
        # Обновить _ для текущего ответа
        _ = lambda key, **kw: t(key, detected, **kw)
    
    await message.answer(_("welcome", name=message.from_user.first_name))
```

## Переводы в файлах (Babel/gettext)

Для больших проектов — отраслевой стандарт gettext:

```bash
pip install babel aiogram-i18n
```

Структура:
```
locales/
├── ru/
│   └── LC_MESSAGES/
│       ├── bot.po    # исходник переводов
│       └── bot.mo    # скомпилированный файл
├── en/
│   └── LC_MESSAGES/
│       ├── bot.po
│       └── bot.mo
```

```python
# aiogram_i18n вариант
from aiogram_i18n import I18nMiddleware
from aiogram_i18n.cores import FluentRuntimeCore

i18n = I18nMiddleware(
    core=FluentRuntimeCore(path="locales/{locale}"),
    default_locale="ru"
)
```

## Промпт для добавления i18n в бота

```
Добавь мультиязычность в Telegram-бота на aiogram 3.
Поддержать языки: русский (ru), английский (en).

Реализация:
- Словарь TRANSLATIONS в translations.py (все строки интерфейса)
- SQLite таблица user_settings для хранения lang пользователя
- Middleware I18nMiddleware — добавляет `lang` и `_()` в data хэндлеров
- /lang команда с inline кнопками выбора языка
- Автоопределение при /start из message.from_user.language_code

Переведи все строки интерфейса (welcome, help, кнопки, ошибки).
```

---

::: info Флаги в кнопках
Флаги-эмодзи (`🇷🇺`, `🇬🇧`) выглядят хорошо в кнопках. Но проверь что они корректно отображаются на целевых устройствах — на некоторых системах могут показываться как буквы.
:::
